        const anecdoteCard = document.getElementById('anecdote-card');
        const anecdoteTextContainer = document.getElementById('anecdote-text-container');
        const nextButton = document.getElementById('next-button');
        const shareFacebookBtn = document.getElementById('share-facebook');
        const shareXBtn = document.getElementById('share-x');
        const topicInput = document.getElementById('topic-input');

        let currentAnecdote = "";

        // Function to get a random anecdote from the default topic
        function getRandomAnecdote() {
            const anecdotesForTopic = anecdotesByTopic["everyday_life"];
            const randomIndex = Math.floor(Math.random() * anecdotesForTopic.length);
            return anecdotesForTopic[randomIndex];
        }

        // Function to update the anecdote displayed on the page
        function updateAnecdote() {
            anecdoteTextContainer.style.opacity = 0;
            anecdoteTextContainer.style.transform = 'scale(0.95)';
            setTimeout(() => {
                const requestedTopic = topicInput.value.trim();
                if (requestedTopic !== "") {
                    currentAnecdote = `Thank you for the suggestion! We'll work on an anecdote about "${requestedTopic}". Check back soon!`;
                    topicInput.value = ""; // Clear the input after displaying the message
                } else {
                    currentAnecdote = getRandomAnecdote();
                }
                anecdoteTextContainer.textContent = currentAnecdote;
                anecdoteTextContainer.style.opacity = 1;
                anecdoteTextContainer.style.transform = 'scale(1)';
            }, 300); // Wait for the fade-out transition to complete
        }

        // Function to handle social media sharing
        function shareAnecdote(platform) {
            if (!currentAnecdote || currentAnecdote.startsWith("Thank you")) return; // Don't share if there's no anecdote or it's a thank you message

            const encodedText = encodeURIComponent(`A little moment of joy for you! ðŸ˜Š\n\n"${currentAnecdote}"\n\n#DailySmileApp #Humor #FeelGood`);
            let url = "";

            switch (platform) {
                case 'facebook':
                    // Uses the Facebook Sharer. Note that the 'u' parameter is the URL to share.
                    url = `https://www.facebook.com/sharer/sharer.php?quote=${encodedText}`;
                    break;
                case 'x':
                    url = `https://twitter.com/intent/tweet?text=${encodedText}`;
                    break;
            }

            if (url) {
                window.open(url, '_blank', 'width=600,height=400');
            }
        }
        
        // Display an anecdote when the page loads
        document.addEventListener('DOMContentLoaded', updateAnecdote);
